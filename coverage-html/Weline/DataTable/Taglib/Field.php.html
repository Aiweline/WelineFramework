453    {
454        $target = [

457            'alias' => null,

460
461        $modelAliasMap = self::extractModelAliasMap($defaultModelClass, $tableContext);
462        if (!empty($modelAliasMap['main'])) {
463            $target['model'] = $modelAliasMap['main'];

466        if (strpos($fieldName, '.') === false) {

478        }

483    private static function extractModelAliasMap(string $modelClass, ?array $tableContext): array
484    {
485        $result = [
486            'main' => '',
487            'models' => [],
488        ];
489
490        if (!empty($tableContext['model_config']['models']) && is_array($tableContext['model_config']['models'])) {
491            $result['models'] = $tableContext['model_config']['models'];
492            $result['main'] = $tableContext['model_config']['main_model'] ?? (reset($result['models']) ?: '');
493            return $result;
494        }
495
496        if (strpos($modelClass, ',') === false) {
497            $result['main'] = $modelClass;
498            return $result;
499        }
500
501        $parts = array_map('trim', explode(',', $modelClass));
502        foreach ($parts as $part) {
503            if (preg_match('/^(.+?)\s+as\s+([A-Za-z_][A-Za-z0-9_]*)$/i', $part, $matches)) {
504                $resolvedModel = trim($matches[1]);
505                $alias = trim($matches[2]);
506            } else {
507                $resolvedModel = trim($part);
508                $alias = basename(str_replace('\\', '/', $resolvedModel));
509            }
510
511            if (empty($result['main'])) {
512                $result['main'] = $resolvedModel;
513            }
514
515            $result['models'][$alias] = $resolvedModel;
516        }
517
518        return $result;
519    }
