510    // =============== 预定义的状态重置 ===============
511    
512    /**
513     * 注册框架核心重置回调
514     * 
515     * 原则：FPM 下每次请求都是全新状态，WLS 下必须手动重置到同等状态。
516     * 所有持有请求级数据的静态变量和递归保护标志，都必须在此注册。
517     * 
518     * 分类：
519     * 1. URL/路由解析缓存 — 每次请求 URL 不同，必须清除
520     * 2. 递归保护标志 — FPM 下每次都是 false，WLS 下如果异常中断可能残留 true
521     * 3. 请求上下文 — RequestContext、SSE 连接等请求级状态
522     * 4. 用户/会话状态 — Env::$user 等与当前请求用户绑定的数据
523     */
524    public static function registerFrameworkResets(): void
525    {
526        // ========== 1. URL / 路由解析缓存 ==========
527        
528        // Url 类：parserServer, parserSites, parserCurrencies, parserLanguages,
529        //         parserMatchs, parserSiteMatchs, parserUrlCache, splitUrlCache,
530        //         parserCache, parsingInProgress, decode_urls
531        \Weline\Framework\Http\Url::registerStateResets();
532        
533        // Request::$url_paths — URL 路径解析缓存
534        self::registerStaticReset(\Weline\Framework\Http\Request::class, 'url_paths', []);
535        
536        // ProcessUrlCache — 请求内 URL 缓存
537        // $staticCache 是 private，反射扫描 public 属性会遗漏。
538        // 此缓存标注为"请求内静态缓存"，FPM 下每次请求自动归零，
539        // WLS 下必须显式重置，否则上个请求的路由缓存会泄漏到下个请求。
540        self::registerResetCallback('process_url_cache_static', function () {
541            try {
542                $ref = new \ReflectionClass(\Weline\Framework\Router\Cache\ProcessUrlCache::class);
543                if ($ref->hasProperty('staticCache')) {
544                    $prop = $ref->getProperty('staticCache');
545                    $prop->setAccessible(true);
546                    $prop->setValue(null, null);
547                }
548            } catch (\Throwable $e) {
549                // 忽略反射错误
550            }
551        });
552        
553        // ========== 2. 递归保护标志 ==========
554        // FPM 下这些标志每次请求都是 false。
555        // WLS 下如果请求处理过程中抛出异常，标志可能残留为 true，
556        // 导致下一个请求跳过关键逻辑（创建 Session、创建缓存驱动等）。
557        
558        // CacheFactory::$isCreating — 防止缓存工厂创建时循环调用
559        self::registerStaticReset(\Weline\Framework\Cache\CacheFactory::class, 'isCreating', false);
560        
561        // CacheFactory::$creatingDriver — 防止缓存驱动创建时循环调用
562        self::registerStaticReset(\Weline\Framework\Cache\CacheFactory::class, 'creatingDriver', false);
563        
564        // Taglib::$compileDepth — 嵌套编译深度，异常中断可能残留非零值
565        self::registerStaticReset(\Weline\Framework\View\Taglib::class, 'compileDepth', 0);
566        
567        // ResultManager::$result — 控制器 success/error/info 结果，请求级，redirect 后消费
568        \Weline\Framework\Manager\ResultManager::registerStateResets();
569        
570        // ========== 3. 请求上下文 ==========
571        
572        // RequestContext — 请求 ID、区域、语言、货币等请求级上下文
573        self::registerResetCallback('request_context', function () {
574            RequestContext::cleanup();
575        });
576        
577        // SseContext — SSE 连接、启用标志、回调等请求级状态
578        self::registerResetCallback('sse_context', function () {
579            if (\class_exists(\Weline\Framework\Http\Sse\SseContext::class, false)) {
580                \Weline\Framework\Http\Sse\SseContext::reset();
581            }
582        });
583        
584        // ========== 4. 用户 / 会话状态 ==========
585        
586        // Env::$user — 当前请求的用户标识，不能泄漏到下一个请求
587        self::registerStaticReset(\Weline\Framework\App\Env::class, 'user', '');
588        
589        // Session 实例清理 — WLS 下 SessionFactory 缓存的 Session 实例是请求级状态。
590        // 不重置会导致：不同用户的请求共用同一个 Session ID（安全泄漏），
591        //               或重载后新 Cookie 无法被读取（session 丢失）。
592        self::registerResetCallback('session_instances', function () {
593            // 清理 SessionFactory 缓存的请求级实例
594            if (\class_exists(\Weline\Framework\Session\SessionFactory::class, false)) {
595                \Weline\Framework\Session\SessionFactory::getInstance()->resetRequestInstances();
596            }
597        });
598        
599        // 缓存系统已重构为 w_cache() 全局函数，不再使用旧的 XxxCache 类
600        // 缓存池的清理由 CachePoolInterface 统一管理
601        
602        // ========== 5. 请求级对象实例清理 ==========
603        // FPM 下 Response 每次请求都是新实例。
604        // WLS 下 ObjectManager 缓存的 Response 可能残留上个请求的 body/headers。
605        
606        self::registerResetCallback('request_scoped_objects', function () {
607            \Weline\Framework\Manager\ObjectManager::removeInstance(\Weline\Framework\Http\Response::class);
608        });
609        
610        // Template 单例实例清理 — WLS 下 Template 使用 static 单例，init() 仅首次创建时调用。
611        // _data 数组中的 title、req、env、local 以及 view_dir、template_dir 等目录路径
612        // 全部是请求级数据，会残留到下一个请求，导致页面标题错乱、模板路径指向上个模块。
613        self::registerResetCallback('template_instance', function () {
614            \Weline\Framework\View\Template::resetInstance();
615            \Weline\Framework\Manager\ObjectManager::removeInstance(\Weline\Framework\View\Template::class);
616        });
617        
618        // State 实例清理 — State::$is_backend 在构造函数中根据当前请求设置。
619        // WLS 下 State 被 ObjectManager 缓存后构造函数不再调用，$is_backend 可能残留上个请求的值。
620        // 重置 static 变量 + 清除实例，确保下个请求重新创建 State 并正确判断 backend/frontend。
621        self::registerStaticReset(\Weline\Framework\App\State::class, 'is_backend', false);
622        self::registerStaticReset(\Weline\Framework\App\State::class, 'langLocalCache', null);
623        self::registerResetCallback('state_instance', function () {
624            \Weline\Framework\Manager\ObjectManager::removeInstance(\Weline\Framework\App\State::class);
625        });
626        
627        // 控制器实例清理 — WLS 下控制器被 ObjectManager 缓存为单例。
628        // 控制器的 __init() 只在首次创建时调用，导致 $this->request 指向旧请求对象。
629        // 例如：第一个 GET 请求创建了控制器，$this->request = WlsRequest(GET)；
630        //       第二个 POST 请求复用了缓存的控制器，$this->request 仍然是 GET 的，
631        //       于是 isPost() 返回 false → "无效的请求方法"。
632        // 解决：每次请求开始时清理控制器缓存，确保 __init() 重新执行获取最新 Request。
633        self::registerResetCallback('controller_instances', function () {
634            $instances = \Weline\Framework\Manager\ObjectManager::getInstances();
635            foreach ($instances as $class => $instance) {
636                if ($instance instanceof \Weline\Framework\Controller\Core) {
637                    \Weline\Framework\Manager\ObjectManager::removeInstance($class);
638                }
639            }
640        });
641        
642        // ========== 5.5 模型实例清理 ==========
643        // WLS 下模型被 ObjectManager 缓存为单例，注入到控制器后跨请求复用。
644        // 模型内部持有 items、_data、_model_fields_data 等请求级数据，
645        // 如果不清理，上一请求的数据（包括主键 ID）会残留到下一请求。
646        // 例如：第一个请求 save() 成功设置了 user_id = 5，
647        //       第二个请求复用同一个模型实例，clearData() 不清理 _data 的主键，
648        //       导致新增变成更新，或 unique constraint violation。
649        // 解决：每次请求结束时清理所有 AbstractModel 子类的缓存实例。
650        self::registerResetCallback('model_instances', function () {
651            $instances = \Weline\Framework\Manager\ObjectManager::getInstances();
652            foreach ($instances as $class => $instance) {
653                if ($instance instanceof \Weline\Framework\Database\AbstractModel) {
654                    \Weline\Framework\Manager\ObjectManager::removeInstance($class);
655                }
656            }
657        });
658        
659        // ========== 5.6 Request 实例清理 ==========
660        // WLS 下 Request 被 ObjectManager 缓存复用，其实例属性 request_id 在首次
661        // getId() 调用后不会重置，导致多个请求使用相同的 request_id。
662        // BackendActivityLog 等使用 request_id 作为唯一约束的表会触发冲突。
663        // 解决：每次请求结束后清除 Request 实例，确保下次请求重新创建。
664        self::registerResetCallback('request_instance', function () {
665            \Weline\Framework\Manager\ObjectManager::removeInstance(\Weline\Framework\Http\Request::class);
666        });
667        
668        // ========== 5.7 Observer 实例清理 ==========
669        // WLS 下 Observer 被 ObjectManager 缓存为单例，其构造函数注入的 Request 对象
670        // 指向旧请求实例。虽然 WlsRuntime::handle() 会将新 WlsRequest 注册到 ObjectManager，
671        // 但已缓存的 Observer 仍持有旧 Request 引用，导致 request_id 重复。
672        // 例如：BackendControllerInit Observer 调用 $this->request->getId() 获取的是
673        //       上一个请求的 request_id，导致 BackendActivityLog 唯一约束冲突。
674        // 解决：每次请求结束时清理所有 ObserverInterface 实现类的缓存实例，
675        //       确保下次请求重新创建 Observer 并注入最新的 Request。
676        self::registerResetCallback('observer_instances', function () {
677            $instances = \Weline\Framework\Manager\ObjectManager::getInstances();
678            foreach ($instances as $class => $instance) {
679                if ($instance instanceof \Weline\Framework\Event\ObserverInterface) {
680                    \Weline\Framework\Manager\ObjectManager::removeInstance($class);
681                }
682            }
683        });
684        
685        // ========== 5.8 Router 实例清理 ==========
686        // WLS 下 Router\Core 被 ObjectManager 缓存为单例。虽然 __init() 有 $isNewRequest 检测，
687        // 但 ObjectManager 从缓存返回实例时不会再次调用 __init()，导致：
688        // 1. $this->request 仍指向旧请求对象
689        // 2. $this->area_router、$this->is_backend 等属性未更新
690        // 3. 后台请求可能使用前端 area_router，导致 404 错误
691        // 解决：每次请求结束时清理 Router\Core 实例，确保下次请求重新创建并正确初始化。
692        self::registerResetCallback('router_core_instance', function () {
693            \Weline\Framework\Manager\ObjectManager::removeInstance(\Weline\Framework\Router\Core::class);
694        });
695        
696        // ========== 6. 数据库连接事务清理 ==========

699        
700        self::registerResetCallback('db_connection_cleanup', function () {
