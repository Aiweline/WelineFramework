235     *

243        // 如果名称已经是标准格式（包含破折号），直接返回小写版本
244        if (strpos($name, '-') !== false && strpos($name, '_') === false) {
245            return strtolower($name);
246        }
247        
248        // 清理组件名称：移除路径分隔符，统一为破折号格式

250        // 移除多余的破折号

254        // 如果名称以 category 开头，移除重复的前缀

256        $categoryPrefix = strtolower($category) . '-';
257        if (stripos($cleanName, $categoryPrefix) === 0) {
258            $cleanName = substr($cleanName, strlen($categoryPrefix));
259        }
260        
261        // 如果名称等于 category，保持原名
262        if (strtolower($cleanName) === strtolower($category)) {
263            $cleanName = $category;
264        }
265        
266        // 如果名称为空，使用 category 作为名称
267        if (empty($cleanName)) {
268            $cleanName = $category;
269        }
270        
271        // 生成组件代码：{category}-{name}（不带模板前缀）

651        $componentModel = \Weline\Framework\Manager\ObjectManager::getInstance(self::class);
