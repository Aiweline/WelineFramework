# DataTable 示例代码

## 基础示例

### 1. 最简单的表格

```phtml
<!-- 最基本的用法，自动生成所有字段 -->
<w:d-table model="App\Model\User" scope="user-basic"></w:d-table>
```

### 2. 带分页的表格

```phtml
<w:d-table model="App\Model\User" 
           scope="user-pagination" 
           page-size="15" 
           show-pagination="true">
</w:d-table>
```

### 3. 自定义表头

```phtml
<w:d-table model="App\Model\User" scope="user-custom-header">
    <w:t-header>
        <w:field belong="t-header" name="id" sortable="true" width="80">用户ID</w:field>
        <w:field belong="t-header" name="username" sortable="true" width="150">用户名</w:field>
        <w:field belong="t-header" name="email" sortable="true" width="200">邮箱地址</w:field>
        <w:field belong="t-header" name="status" sortable="true" width="100">状态</w:field>
        <w:field belong="t-header" name="created_at" sortable="true" width="150">创建时间</w:field>
    </w:t-header>
</w:d-table>
```

## 筛选功能示例

### 1. 基础筛选

```phtml
<w:d-table model="App\Model\User" scope="user-filter" searchable="true">
    <w:t-filter>
        <w:field belong="t-filter" name="username" type="text" placeholder="搜索用户名"></w:field>

    </w:t-filter>
```


## JavaScript 集成示例

### 1. 自定义事件处理

```javascript
// 监听表格数据加载完成事件
document.addEventListener('datatable:loaded', function(e) {
    const tableId = e.detail.tableId;
    const data = e.detail.data;
    console.log('表格加载完成:', tableId, data);
});

// 监听记录保存事件
document.addEventListener('datatable:saved', function(e) {
    const tableId = e.detail.tableId;
    const record = e.detail.record;
    console.log('记录保存成功:', tableId, record);
    
    // 可以在这里添加自定义逻辑
    alert('保存成功！');
});

// 监听记录删除事件
document.addEventListener('datatable:deleted', function(e) {
    const tableId = e.detail.tableId;
    const recordId = e.detail.recordId;
    console.log('记录删除成功:', tableId, recordId);
});
```

### 2. 程序化操作

```javascript
// 刷新表格数据
DataTableManager.refreshTable('user-management');

// 打开新增表单
DataTableFormManager.openModal('user-form', 'add');

// 打开编辑表单
DataTableFormManager.openModal('user-form', 'edit', 123);

// 获取表格当前数据
const tableData = DataTableManager.getTableData('user-management');

// 设置筛选条件
DataTableManager.setFilters('user-management', {
    username: '张三',
    status: '1'
});
```
