## 0. 核心决策：WeAsync

**WeAsync** 是 Weline Framework 自研的高性能异步引擎，**完全照搬 Workerman 开源代码**，改名后作为框架的异步运行时。

### 为什么照搬 Workerman

| 考量 | 说明 |
|------|------|
| **成熟稳定** | Workerman 经过 10+ 年生产验证 |
| **MIT 协议** | 完全开源，可自由使用和修改 |
| **自主可控** | 照搬后可自行维护和优化 |
| **性能保证** | 直接获得 50,000+ QPS 的性能 |

### 照搬范围

```
Workerman 源码 → WeAsync 模块
├── Worker.php           → Worker.php
├── Connection/          → Connection/
├── Events/              → Event/
├── Protocols/           → Protocol/
└── Lib/Timer.php        → Timer.php
```

---

## 1. 设计目标

### 1.1 核心目标

| 目标 | 描述 |
|------|------|
| **运行时透明** | 业务代码无需关心运行在 FPM 还是 WeAsync |
| **状态隔离** | 每个请求的状态完全隔离，无污染 |
| **高性能** | WeAsync 模式下达到 50,000+ QPS |
| **渐进迁移** | 现有代码无需大改即可运行 |

### 1.2 非目标

- ❌ 不重新发明轮子（直接照搬 Workerman）
- ❌ 不强制异步 I/O（保持同步 API 可用）
- ❌ 不破坏现有模块兼容性

### 2.2 请求生命周期对比

#### FPM 模式

```
请求1: [init] → [bootstrap] → [handle] → [terminate] → [destroy]
请求2: [init] → [bootstrap] → [handle] → [terminate] → [destroy]
请求3: [init] → [bootstrap] → [handle] → [terminate] → [destroy]
        ↑                                                    ↑
        └─────────── 每次请求都完整执行 ──────────────────────┘
```

#### 常驻内存模式

```
启动:   [init] → [bootstrap]
                    ↓
请求1:           [handle] → [reset]
请求2:           [handle] → [reset]
请求3:           [handle] → [reset]
                    ↓
关闭:            [terminate] → [destroy]
```

## 3. 核心组件设计

### 3.1 RuntimeInterface

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Message\ResponseInterface;

interface RuntimeInterface
{
    /**
     * 启动运行时（只执行一次）
     * 
     * - 加载配置
     * - 初始化 DI 容器
     * - 注册服务提供者
     * - 预热缓存
     */
    public function bootstrap(): void;
    
    /**
     * 处理单个请求
     * 
     * - 创建请求上下文
     * - 路由分发
     * - 执行控制器
     * - 返回响应
     */
    public function handle(ServerRequestInterface $request): ResponseInterface;
    
    /**
     * 请求结束后重置状态
     * 
     * - 清理请求级实例
     * - 重置静态变量
     * - 释放临时资源
     */
    public function reset(): void;
    
    /**
     * 关闭运行时
     * 
     * - 关闭数据库连接
     * - 清理资源
     * - 保存状态
     */
    public function terminate(): void;
    
    /**
     * 获取运行时类型
     */
    public function getType(): RuntimeType;
    
    /**
     * 是否为常驻内存模式
     */
    public function isPersistent(): bool;
}
```

### 3.2 RuntimeType 枚举

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

enum RuntimeType: string
{
    case FPM = 'fpm';
    case CLI = 'cli';
    case WEASYNC = 'weasync';  // WeAsync 自研异步引擎
    
    /**
     * 是否为常驻内存模式
     */
    public function isPersistent(): bool
    {
        return match($this) {
            self::WEASYNC => true,
            default => false,
        };
    }
    
    /**
     * 是否支持多进程
     */
    public function supportsMultiProcess(): bool
    {
        return match($this) {
            self::WEASYNC => true,
            default => false,
        };
    }
}
```

### 3.3 RequestContext（请求上下文）

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

use Fiber;
use WeakMap;

/**
 * 请求上下文容器
 * 
 * 职责：
 * - 存储当前请求的所有状态
 * - 隔离不同请求的数据
 * - 支持协程环境
 */
final class RequestContext
{
    /**
     * 使用 Fiber 本地存储（协程安全）
     */
    private static ?WeakMap $fiberContexts = null;
    
    /**
     * 非协程环境的上下文存储
     */
    private static ?self $currentContext = null;
    
    /**
     * 请求唯一标识
     */
    public private(set) readonly string $requestId;
    
    /**
     * 请求开始时间
     */
    public private(set) readonly float $startTime;
    
    /**
     * 请求级实例存储
     */
    private array $instances = [];
    
    /**
     * 请求属性
     */
    private array $attributes = [];
    
    private function __construct()
    {
        $this->requestId = bin2hex(random_bytes(16));
        $this->startTime = microtime(true);
    }
    
    /**
     * 获取当前请求上下文
     */
    public static function current(): self
    {
        // 协程环境：使用 Fiber 上下文
        $fiber = Fiber::getCurrent();
        if ($fiber !== null) {
            self::$fiberContexts ??= new WeakMap();
            return self::$fiberContexts[$fiber] ??= new self();
        }
        
        // 非协程环境：使用静态变量
        return self::$currentContext ??= new self();
    }
    
    /**
     * 创建新的请求上下文
     */
    public static function create(): self
    {
        $context = new self();
        
        $fiber = Fiber::getCurrent();
        if ($fiber !== null) {
            self::$fiberContexts ??= new WeakMap();
            self::$fiberContexts[$fiber] = $context;
        } else {
            self::$currentContext = $context;
        }
        
        return $context;
    }
    
    /**
     * 销毁当前请求上下文
     */
    public static function destroy(): void
    {
        $fiber = Fiber::getCurrent();
        if ($fiber !== null && self::$fiberContexts !== null) {
            unset(self::$fiberContexts[$fiber]);
        } else {
            self::$currentContext = null;
        }
    }
    
    /**
     * 存储请求级实例
     */
    public function setInstance(string $class, object $instance): void
    {
        $this->instances[$class] = $instance;
    }
    
    /**
     * 获取请求级实例
     */
    public function getInstance(string $class): ?object
    {
        return $this->instances[$class] ?? null;
    }
    
    /**
     * 检查实例是否存在
     */
    public function hasInstance(string $class): bool
    {
        return isset($this->instances[$class]);
    }
    
    /**
     * 设置属性
     */
    public function setAttribute(string $key, mixed $value): void
    {
        $this->attributes[$key] = $value;
    }
    
    /**
     * 获取属性
     */
    public function getAttribute(string $key, mixed $default = null): mixed
    {
        return $this->attributes[$key] ?? $default;
    }
    
    /**
     * 获取请求耗时（毫秒）
     */
    public function getElapsedMs(): float
    {
        return (microtime(true) - $this->startTime) * 1000;
    }
    
    /**
     * 清理所有状态
     */
    public function clear(): void
    {
        $this->instances = [];
        $this->attributes = [];
    }
}
```

## 4. 实例生命周期

### 4.1 实例分类

| 类型 | 生命周期 | 示例 | 存储位置 |
|------|---------|------|---------|
| **全局单例** | 应用级 | `EventsManager`, `Router`, `Config` | `ObjectManager::$instances` |
| **请求单例** | 请求级 | `Request`, `Response`, `Session` | `RequestContext::$instances` |
| **瞬态实例** | 每次创建 | `DataObject`, `Collection` | 不缓存 |
| **连接池** | 应用级（可复用） | `PDO`, `Redis` | `ConnectionPool` |

### 4.2 实例标记

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Manager;

use Attribute;

/**
 * 标记实例的生命周期范围
 */
#[Attribute(Attribute::TARGET_CLASS)]
class InstanceScope
{
    public function __construct(
        public readonly Scope $scope = Scope::SINGLETON,
    ) {}
}

enum Scope: string
{
    /**
     * 全局单例：应用生命周期内只有一个实例
     */
    case SINGLETON = 'singleton';
    
    /**
     * 请求单例：每个请求一个实例
     */
    case REQUEST = 'request';
    
    /**
     * 瞬态：每次获取都创建新实例
     */
    case TRANSIENT = 'transient';
    
    /**
     * 池化：从连接池获取
     */
    case POOLED = 'pooled';
}
```

## 5. 状态管理策略

### 5.1 状态分类

```
┌─────────────────────────────────────────────────────────────┐
│                        状态分类                              │
├─────────────────────────────────────────────────────────────┤
│                                                              │
│  ┌─────────────────────────────────────────────────────┐    │
│  │              不可变状态 (Immutable)                   │    │
│  │  - 配置文件内容                                       │    │
│  │  - 路由表                                            │    │
│  │  - 事件注册表                                         │    │
│  │  → 启动时加载，运行期间不变                            │    │
│  └─────────────────────────────────────────────────────┘    │
│                                                              │
│  ┌─────────────────────────────────────────────────────┐    │
│  │              可变共享状态 (Shared Mutable)            │    │
│  │  - 缓存数据                                          │    │
│  │  - 统计计数器                                         │    │
│  │  - 连接池状态                                         │    │
│  │  → 需要原子操作或锁保护                               │    │
│  └─────────────────────────────────────────────────────┘    │
│                                                              │
│  ┌─────────────────────────────────────────────────────┐    │
│  │              请求隔离状态 (Request Scoped)            │    │
│  │  - 当前用户                                          │    │
│  │  - 请求参数                                          │    │
│  │  - Session 数据                                      │    │
│  │  → 存储在 RequestContext，请求结束后清理              │    │
│  └─────────────────────────────────────────────────────┘    │
│                                                              │
└─────────────────────────────────────────────────────────────┘
```

### 5.2 StateManager 接口

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

/**
 * 状态管理器
 * 
 * 负责追踪和重置所有可变状态
 */
interface StateManagerInterface
{
    /**
     * 注册可重置的状态持有者
     */
    public function register(ResettableInterface $resettable): void;
    
    /**
     * 重置所有请求级状态
     */
    public function resetRequestState(): void;
    
    /**
     * 重置所有状态（包括应用级）
     */
    public function resetAll(): void;
    
    /**
     * 获取状态快照（用于调试）
     */
    public function getStateSnapshot(): array;
}

/**
 * 可重置接口
 * 
 * 所有持有可变状态的类应实现此接口
 */
interface ResettableInterface
{
    /**
     * 重置到初始状态
     */
    public function reset(): void;
    
    /**
     * 获取状态信息（用于调试）
     */
    public function getStateInfo(): array;
}
```
