# 04 - 请求响应抽象

> **优先级**: ⭐⭐⭐⭐  
> **依赖**: 02-运行时抽象层, 03-ObjectManager改造  
> **预计工作量**: 2-3 天

---

## 0. 与 WeAsync 的关系

**WeAsync**（照搬 Workerman）有自己的 `Request` 和 `Response` 类，需要与框架的 PSR-7 请求响应对接：

```
WeAsync 请求 → 转换 → 框架 Request（PSR-7）→ 处理 → 框架 Response → 转换 → WeAsync 响应
```

WeAsync 的 `Protocol\Http.php`（照搬自 Workerman）已经实现了 HTTP 解析，我们只需要做转换层。

---

## 1. 概述

为了支持 FPM 和 WeAsync 双模式，需要将请求/响应抽象为 PSR-7 兼容接口。

### 1.1 当前问题

```php
// 当前实现依赖超全局变量
$_GET, $_POST, $_SERVER, $_COOKIE, $_FILES

// 问题1：WeAsync 模式下这些变量在请求间不会自动重置
// 问题2：WeAsync 的请求数据来自 TcpConnection，不是超全局变量
```

### 1.2 改造目标

| 目标 | 描述 |
|------|------|
| **PSR-7 兼容** | 支持标准 HTTP 消息接口 |
| **运行时无关** | 不直接依赖超全局变量 |
| **WeAsync 集成** | 能从 WeAsync Request 转换 |
| **向后兼容** | 保持现有 API 可用 |

## 2. PSR-7 适配

### 2.1 Request 改造

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Http;

use Psr\Http\Message\ServerRequestInterface;
use Weline\Framework\Manager\Attribute\InstanceScope;
use Weline\Framework\Manager\Scope;
use Weline\Framework\Runtime\RequestContext;

#[InstanceScope(Scope::REQUEST)]
class Request
{
    /**
     * 内部 PSR-7 请求对象
     */
    private ?ServerRequestInterface $psrRequest = null;
    
    /**
     * 请求参数缓存
     */
    private array $queryParams = [];
    private array $postParams = [];
    private array $serverParams = [];
    private array $cookieParams = [];
    private array $uploadedFiles = [];
    private ?array $parsedBody = null;
    
    /**
     * 是否已初始化
     */
    private bool $initialized = false;
    
    /**
     * 从 PSR-7 请求初始化
     */
    public function initFromPsr7(ServerRequestInterface $request): self
    {
        $this->psrRequest = $request;
        $this->queryParams = $request->getQueryParams();
        $this->postParams = (array) $request->getParsedBody();
        $this->serverParams = $request->getServerParams();
        $this->cookieParams = $request->getCookieParams();
        $this->uploadedFiles = $request->getUploadedFiles();
        $this->parsedBody = $request->getParsedBody();
        $this->initialized = true;
        
        return $this;
    }
    
    /**
     * 从超全局变量初始化（FPM 兼容模式）
     */
    public function initFromGlobals(): self
    {
        $this->queryParams = $_GET;
        $this->postParams = $_POST;
        $this->serverParams = $_SERVER;
        $this->cookieParams = $_COOKIE;
        $this->uploadedFiles = $_FILES;
        $this->initialized = true;
        
        return $this;
    }
    
    /**
     * 确保已初始化
     */
    private function ensureInitialized(): void
    {
        if (!$this->initialized) {
            $this->initFromGlobals();
        }
    }
    
    /**
     * 获取 GET 参数
     */
    public function getGet(string $key = '', mixed $default = null): mixed
    {
        $this->ensureInitialized();
        
        if ($key === '') {
            return $this->queryParams;
        }
        
        return $this->queryParams[$key] ?? $default;
    }
    
    /**
     * 获取 POST 参数
     */
    public function getPost(string $key = '', mixed $default = null): mixed
    {
        $this->ensureInitialized();
        
        if ($key === '') {
            return $this->postParams;
        }
        
        return $this->postParams[$key] ?? $default;
    }
    
    /**
     * 获取 SERVER 参数
     */
    public function getServer(string $key = '', mixed $default = null): mixed
    {
        $this->ensureInitialized();
        
        if ($key === '') {
            return $this->serverParams;
        }
        
        return $this->serverParams[$key] ?? $default;
    }
    
    /**
     * 获取请求方法
     */
    public function getMethod(): string
    {
        $this->ensureInitialized();
        return $this->serverParams['REQUEST_METHOD'] ?? 'GET';
    }
    
    /**
     * 获取请求 URI
     */
    public function getUri(): string
    {
        $this->ensureInitialized();
        return $this->serverParams['REQUEST_URI'] ?? '/';
    }
    
    /**
     * 获取请求头
     */
    public function getHeader(string $name): ?string
    {
        $this->ensureInitialized();
        
        // PSR-7 方式
        if ($this->psrRequest !== null) {
            $values = $this->psrRequest->getHeader($name);
            return $values[0] ?? null;
        }
        
        // 从 $_SERVER 获取
        $key = 'HTTP_' . strtoupper(str_replace('-', '_', $name));
        return $this->serverParams[$key] ?? null;
    }
    
    /**
     * 获取所有请求头
     */
    public function getHeaders(): array
    {
        $this->ensureInitialized();
        
        if ($this->psrRequest !== null) {
            return $this->psrRequest->getHeaders();
        }
        
        $headers = [];
        foreach ($this->serverParams as $key => $value) {
            if (str_starts_with($key, 'HTTP_')) {
                $name = str_replace('_', '-', substr($key, 5));
                $headers[$name] = [$value];
            }
        }
        
        return $headers;
    }
    
    /**
     * 获取请求体
     */
    public function getBody(): string
    {
        if ($this->psrRequest !== null) {
            return (string) $this->psrRequest->getBody();
        }
        
        return file_get_contents('php://input') ?: '';
    }
    
    /**
     * 获取 JSON 请求体
     */
    public function getJson(): array
    {
        $body = $this->getBody();
        if ($body === '') {
            return [];
        }
        
        try {
            return json_decode($body, true, 512, JSON_THROW_ON_ERROR);
        } catch (\JsonException) {
            return [];
        }
    }
    
    /**
     * 是否为 AJAX 请求
     */
    public function isAjax(): bool
    {
        return $this->getHeader('X-Requested-With') === 'XMLHttpRequest';
    }
    
    /**
     * 是否为 JSON 请求
     */
    public function isJson(): bool
    {
        $contentType = $this->getHeader('Content-Type') ?? '';
        return str_contains($contentType, 'application/json');
    }
    
    /**
     * 获取客户端 IP
     */
    public function getClientIp(): string
    {
        $this->ensureInitialized();
        
        // 检查代理头
        $headers = ['HTTP_X_FORWARDED_FOR', 'HTTP_CLIENT_IP', 'REMOTE_ADDR'];
        
        foreach ($headers as $header) {
            if (!empty($this->serverParams[$header])) {
                $ips = explode(',', $this->serverParams[$header]);
                return trim($ips[0]);
            }
        }
        
        return '';
    }
    
    /**
     * 获取底层 PSR-7 请求（如果有）
     */
    public function getPsrRequest(): ?ServerRequestInterface
    {
        return $this->psrRequest;
    }
}
```

### 2.2 Response 改造

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Http;

use Psr\Http\Message\ResponseInterface;
use Weline\Framework\Manager\Attribute\InstanceScope;
use Weline\Framework\Manager\Scope;

#[InstanceScope(Scope::REQUEST)]
class Response
{
    private int $statusCode = 200;
    private string $reasonPhrase = 'OK';
    private array $headers = [];
    private string $body = '';
    private ?ResponseInterface $psrResponse = null;
    
    /**
     * 设置状态码
     */
    public function setStatusCode(int $code, string $reasonPhrase = ''): self
    {
        $this->statusCode = $code;
        $this->reasonPhrase = $reasonPhrase ?: $this->getDefaultReasonPhrase($code);
        return $this;
    }
    
    /**
     * 获取状态码
     */
    public function getStatusCode(): int
    {
        return $this->statusCode;
    }
    
    /**
     * 设置响应头
     */
    public function setHeader(string $name, string $value): self
    {
        $this->headers[$name] = $value;
        return $this;
    }
    
    /**
     * 获取响应头
     */
    public function getHeaders(): array
    {
        return $this->headers;
    }
    
    /**
     * 设置响应体
     */
    public function setBody(string $body): self
    {
        $this->body = $body;
        return $this;
    }
    
    /**
     * 获取响应体
     */
    public function getBody(): string
    {
        return $this->body;
    }
    
    /**
     * 设置 JSON 响应
     */
    public function json(mixed $data, int $statusCode = 200): self
    {
        $this->statusCode = $statusCode;
        $this->headers['Content-Type'] = 'application/json; charset=utf-8';
        $this->body = json_encode($data, JSON_UNESCAPED_UNICODE);
        return $this;
    }
    
    /**
     * 设置 HTML 响应
     */
    public function html(string $html, int $statusCode = 200): self
    {
        $this->statusCode = $statusCode;
        $this->headers['Content-Type'] = 'text/html; charset=utf-8';
        $this->body = $html;
        return $this;
    }
    
    /**
     * 重定向
     */
    public function redirect(string $url, int $statusCode = 302): self
    {
        $this->statusCode = $statusCode;
        $this->headers['Location'] = $url;
        return $this;
    }
    
    /**
     * 转换为 PSR-7 响应
     */
    public function toPsr7(ResponseInterface $prototype): ResponseInterface
    {
        $response = $prototype
            ->withStatus($this->statusCode, $this->reasonPhrase);
        
        foreach ($this->headers as $name => $value) {
            $response = $response->withHeader($name, $value);
        }
        
        $response->getBody()->write($this->body);
        
        return $response;
    }
    
    /**
     * 发送响应（FPM 模式）
     */
    public function send(): void
    {
        // 发送状态码
        http_response_code($this->statusCode);
        
        // 发送头部
        foreach ($this->headers as $name => $value) {
            header("{$name}: {$value}");
        }
        
        // 发送 body
        echo $this->body;
    }
    
    private function getDefaultReasonPhrase(int $code): string
    {
        return match($code) {
            200 => 'OK',
            201 => 'Created',
            204 => 'No Content',
            301 => 'Moved Permanently',
            302 => 'Found',
            304 => 'Not Modified',
            400 => 'Bad Request',
            401 => 'Unauthorized',
            403 => 'Forbidden',
            404 => 'Not Found',
            405 => 'Method Not Allowed',
            500 => 'Internal Server Error',
            502 => 'Bad Gateway',
            503 => 'Service Unavailable',
            default => '',
        };
    }
}
```

## 3. ServerRequest 工厂

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Http;

use Psr\Http\Message\ServerRequestInterface;
use Nyholm\Psr7\Factory\Psr17Factory;
use Nyholm\Psr7Server\ServerRequestCreator;

/**
 * PSR-7 请求工厂
 */
final class ServerRequestFactory
{
    private static ?Psr17Factory $psr17Factory = null;
    private static ?ServerRequestCreator $creator = null;
    
    /**
     * 从超全局变量创建 PSR-7 请求
     */
    public static function fromGlobals(): ServerRequestInterface
    {
        self::ensureFactory();
        return self::$creator->fromGlobals();
    }
    
    /**
     * 从数组创建 PSR-7 请求
     */
    public static function fromArrays(
        array $server,
        array $headers = [],
        array $cookie = [],
        array $get = [],
        ?array $post = null,
        array $files = [],
        $body = null
    ): ServerRequestInterface {
        self::ensureFactory();
        
        return self::$creator->fromArrays(
            $server,
            $headers,
            $cookie,
            $get,
            $post,
            $files,
            $body
        );
    }
    
    /**
     * 获取 PSR-17 工厂
     */
    public static function getPsr17Factory(): Psr17Factory
    {
        self::ensureFactory();
        return self::$psr17Factory;
    }
    
    private static function ensureFactory(): void
    {
        if (self::$psr17Factory === null) {
            self::$psr17Factory = new Psr17Factory();
            self::$creator = new ServerRequestCreator(
                self::$psr17Factory,
                self::$psr17Factory,
                self::$psr17Factory,
                self::$psr17Factory
            );
        }
    }
}
```

## 4. 超全局变量模拟

### 4.1 GlobalsEmulator

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Http;

use Psr\Http\Message\ServerRequestInterface;

/**
 * 超全局变量模拟器
 * 
 * 用于常驻模式下模拟 $_GET, $_POST 等
 */
final class GlobalsEmulator
{
    /**
     * 从 PSR-7 请求同步到超全局变量
     */
    public static function syncFromPsr7(ServerRequestInterface $request): void
    {
        $_SERVER = $request->getServerParams();
        $_GET = $request->getQueryParams();
        $_POST = (array) $request->getParsedBody();
        $_COOKIE = $request->getCookieParams();
        $_FILES = self::convertUploadedFiles($request->getUploadedFiles());
        $_REQUEST = array_merge($_GET, $_POST, $_COOKIE);
    }
    
    /**
     * 重置超全局变量
     */
    public static function reset(): void
    {
        $_GET = [];
        $_POST = [];
        $_COOKIE = [];
        $_FILES = [];
        $_REQUEST = [];
        $_SERVER = [
            'REQUEST_TIME' => time(),
            'REQUEST_TIME_FLOAT' => microtime(true),
        ];
    }
    
    /**
     * 转换 PSR-7 上传文件格式
     */
    private static function convertUploadedFiles(array $uploadedFiles): array
    {
        $files = [];
        
        foreach ($uploadedFiles as $key => $file) {
            if (is_array($file)) {
                $files[$key] = self::convertUploadedFiles($file);
            } else {
                $files[$key] = [
                    'name' => $file->getClientFilename(),
                    'type' => $file->getClientMediaType(),
                    'tmp_name' => $file->getStream()->getMetadata('uri'),
                    'error' => $file->getError(),
                    'size' => $file->getSize(),
                ];
            }
        }
        
        return $files;
    }
}
```
