# 09 - 异步事件系统

> **优先级**: ⭐⭐⭐

> **预计工作量**: 2-3 天

## 0. 与 WeAsync 的关系

在 **WeAsync**（照搬 Workerman）模式下，可以利用事件循环实现真正的异步事件：

```
FPM 模式下的"异步"：
  事件触发 → 写入队列 → 返回响应 → 后台进程消费队列

WeAsync 模式下的异步：
  事件触发 → 使用 Timer::add() 延迟执行 → 立即返回响应
  或者
  事件触发 → 写入队列 → 其他 Worker 消费
```

WeAsync 的 `Timer` 类（照搬自 Workerman）可以实现非阻塞的延迟执行。

## 1. 概述

框架已有的事件系统支持 `async: true` 标记，需要实现真正的异步执行。

### 1.1 当前状态

```php
// 文档中规划的异步事件注解
#[EventListener('user.deleted', async: true)]
public function onUserDeleted(UserDeletedEvent $event): void {}
```

### 1.2 设计目标

| 目标 | 描述 |
|------|------|
| **非阻塞** | 异步事件不阻塞主请求 |
| **可靠性** | 支持失败重试 |
| **可追踪** | 事件执行状态可查询 |
| **向后兼容** | 默认仍为同步执行 |

<?php
declare(strict_types=1);

namespace Weline\Framework\Event;

enum AsyncStrategy: string
{
    /**
     * 同步执行（默认）
     */
    case SYNC = 'sync';
    
    /**
     * 使用消息队列
     */
    case QUEUE = 'queue';
    
    /**
     * 使用 Fiber 协程
     */
    case FIBER = 'fiber';

     * 使用 defer（请求结束后执行）
     */

}

## 3. 异步事件分发

### 3.1 AsyncEventDispatcher

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Event;

use Weline\Framework\Queue\QueueInterface;
use Weline\Framework\Event\Attribute\EventListener;
use Fiber;

/**
 * 异步事件分发器
 */
final class AsyncEventDispatcher
{
    /**
     * 延迟执行的事件队列
     */
    private static array $deferredEvents = [];
    
    public function __construct(
        private readonly QueueInterface $queue,
    ) {}
    
    /**
     * 异步分发事件
     */
    public function dispatch(
        string $eventName,
        array $data,
        EventListener $config
    ): void {
        $job = new EventJob(
            eventName: $eventName,
            data: $data,
            observerClass: $config->observerClass ?? '',
            maxRetries: $config->maxRetries,
            retryDelay: $config->retryDelay,
        );
        
        match($config->strategy) {
            AsyncStrategy::QUEUE => $this->dispatchToQueue($job, $config->queue),
            AsyncStrategy::FIBER => $this->dispatchToFiber($job),
            AsyncStrategy::DEFER => $this->dispatchToDefer($job),
            default => throw new \RuntimeException('Sync events should not use AsyncEventDispatcher'),
        };
    }
    
    /**
     * 通过消息队列分发
     */
    private function dispatchToQueue(EventJob $job, ?string $queueName): void
    {
        $this->queue->push(
            $job,
            $queueName ?? 'events'
        );
    }
    
    /**
     * 通过 Fiber 分发（立即异步执行）
     */
    private function dispatchToFiber(EventJob $job): void
    {
        $fiber = new Fiber(function() use ($job) {
            $job->execute();
        });
        
        // 启动 Fiber，让事件循环调度
        $fiber->start();
    }
    
    /**
     * 延迟到请求结束后执行
     */
    private function dispatchToDefer(EventJob $job): void
    {
        self::$deferredEvents[] = $job;
    }
    
    /**
     * 执行所有延迟事件（由 Runtime 在请求结束时调用）
     */
    public static function executeDeferredEvents(): void
    {
        foreach (self::$deferredEvents as $job) {
            try {
                $job->execute();
            } catch (\Throwable $e) {
                error_log("Deferred event failed: {$e->getMessage()}");
            }
        }
        
        self::$deferredEvents = [];
    }
}
```

### 3.2 EventJob

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Event;

use Weline\Framework\Queue\JobInterface;
use Weline\Framework\Manager\ObjectManager;

/**
 * 事件作业（用于队列执行）
 */
final class EventJob implements JobInterface
{
    private int $attempts = 0;
    
    public function __construct(
        public readonly string $eventName,
        public readonly array $data,
        public readonly string $observerClass,
        public readonly int $maxRetries = 3,
        public readonly int $retryDelay = 60,
    ) {}
    
    /**
     * 执行事件处理
     */
    public function execute(): void
    {
        $this->attempts++;
        
        try {
            $observer = ObjectManager::getInstance($this->observerClass);
            
            $event = new Event($this->data);
            $event->setName($this->eventName);
            
            $observer->execute($event);
        } catch (\Throwable $e) {
            if ($this->attempts < $this->maxRetries) {
                throw $e; // 让队列处理重试
            }
            
            // 最后一次失败，记录错误
            error_log(sprintf(
                'Event %s failed after %d attempts: %s',
                $this->eventName,
                $this->attempts,
                $e->getMessage()
            ));
        }
    }
    
    /**
     * 获取重试延迟
     */
    public function getRetryDelay(): int
    {
        // 指数退避
        return $this->retryDelay * (2 ** ($this->attempts - 1));
    }
    
    /**
     * 是否应该重试
     */
    public function shouldRetry(): bool
    {
        return $this->attempts < $this->maxRetries;
    }
    
    /**
     * 序列化（用于队列）
     */
    public function serialize(): string
    {
        return json_encode([
            'eventName' => $this->eventName,
            'data' => $this->data,
            'observerClass' => $this->observerClass,
            'maxRetries' => $this->maxRetries,
            'retryDelay' => $this->retryDelay,
            'attempts' => $this->attempts,
        ]);
    }
    
    /**
     * 反序列化
     */
    public static function unserialize(string $payload): self
    {
        $data = json_decode($payload, true);
        
        $job = new self(
            eventName: $data['eventName'],
            data: $data['data'],
            observerClass: $data['observerClass'],
            maxRetries: $data['maxRetries'],
            retryDelay: $data['retryDelay'],
        );
        
        $job->attempts = $data['attempts'] ?? 0;
        
        return $job;
    }
}
```

## 4. EventsManager 改造

```php
<?php
// 在 EventsManager 中集成异步支持

public function dispatch(string $eventName, mixed &$data = []): static
{
    $observers = $this->getEventObservers($eventName);
    
    if (empty($observers)) {
        return $this;
    }
    
    foreach ($observers as $observer) {
        // 检查是否为异步事件
        $isAsync = ($observer['async'] ?? false) === true 
            || ($observer['async'] ?? '') === 'true';
        
        if ($isAsync && $this->asyncDispatcher !== null) {
            // 异步执行
            $this->asyncDispatcher->dispatch(
                $eventName,
                is_array($data) ? $data : ['data' => $data],
                $this->createEventListenerConfig($observer)
            );
        } else {
            // 同步执行
            $this->executeObserver($eventName, $data, $observer);
        }
    }
    
    return $this;
}

private function createEventListenerConfig(array $observer): EventListener
{
    return new EventListener(
        event: $observer['event'] ?? '',
        priority: (int) ($observer['priority'] ?? 0),
        async: true,
        strategy: AsyncStrategy::from($observer['strategy'] ?? 'queue'),
        maxRetries: (int) ($observer['max_retries'] ?? 3),
        retryDelay: (int) ($observer['retry_delay'] ?? 60),
        queue: $observer['queue'] ?? null,
    );
}
```

### 6.1 EventTracker

<?php
declare(strict_types=1);

namespace Weline\Framework\Event;

/**
 * 事件追踪器

final class EventTracker
{
    /**
     * 记录事件分发
     */
    public static function trackDispatch(
        string $eventName,
        string $observerClass,
        AsyncStrategy $strategy
    ): string {
        $trackId = bin2hex(random_bytes(8));
        
        // 记录到日志或数据库
        self::log([
            'track_id' => $trackId,
            'event' => $eventName,
            'observer' => $observerClass,
            'strategy' => $strategy->value,
            'status' => 'dispatched',
            'dispatched_at' => date('Y-m-d H:i:s'),
        ]);
        
        return $trackId;
    }

     * 记录事件完成

            'track_id' => $trackId,

            'error' => $error,
            'failed_at' => date('Y-m-d H:i:s'),
        ]);
    }

    }
}
