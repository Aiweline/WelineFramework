## 0. 与 WeAsync 的关系

**WeAsync**（照搬 Workerman）是常驻内存模式，与 FPM 的运行模型完全不同。本文档确保现有的 FPM 模块能在 WeAsync 模式下正常运行。

### 常见兼容性问题

| 问题 | FPM 行为 | WeAsync 行为 | 解决方案 |
|------|---------|-------------|----------|
| 静态变量 | 请求后销毁 | 跨请求保留 | `#[Resettable]` 注解 |
| `$_GET/$_POST` | 自动设置 | 需要手动注入 | `GlobalsEmulator` |
| `session_start()` | 正常工作 | 会冲突 | 使用 Redis Session |
| 单例模式 | 请求后销毁 | 跨请求复用 | 请求级 Scope |

---

## 1. 概述

确保现有模块在 FPM 和 WeAsync 模式下都能正常运行。

### 1.1 设计目标

| 目标 | 描述 |
|------|------|
| **零修改运行** | 现有模块无需修改即可运行 |
| **渐进优化** | 可选择性地优化特定模块 |
| **明确告警** | 检测到不兼容代码时告警 |

## 2. 兼容性检测

### 2.1 CompatibilityChecker

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

/**
 * 兼容性检测器
 */
final class CompatibilityChecker
{
    private array $warnings = [];
    
    /**
     * 检测类的兼容性
     */
    public function checkClass(string $class): array
    {
        $this->warnings = [];
        
        try {
            $reflection = new \ReflectionClass($class);
            
            // 检查静态属性
            $this->checkStaticProperties($reflection);
            
            // 检查超全局变量使用
            $this->checkSuperGlobalsUsage($reflection);
            
            // 检查单例模式
            $this->checkSingletonPattern($reflection);
            
        } catch (\ReflectionException $e) {
            $this->warnings[] = "Cannot reflect class {$class}: {$e->getMessage()}";
        }
        
        return $this->warnings;
    }
    
    /**
     * 检查静态属性
     */
    private function checkStaticProperties(\ReflectionClass $reflection): void
    {
        foreach ($reflection->getProperties(\ReflectionProperty::IS_STATIC) as $property) {
            // 检查是否标记为 Resettable
            $attributes = $property->getAttributes(Attribute\Resettable::class);
            
            if (empty($attributes)) {
                $this->warnings[] = sprintf(
                    'Static property %s::$%s is not marked as #[Resettable], may cause state leakage in persistent mode',
                    $reflection->getName(),
                    $property->getName()
                );
            }
        }
    }
    
    /**
     * 检查超全局变量使用
     */
    private function checkSuperGlobalsUsage(\ReflectionClass $reflection): void
    {
        $file = $reflection->getFileName();
        if ($file === false) {
            return;
        }
        
        $content = file_get_contents($file);
        
        $superGlobals = ['$_GET', '$_POST', '$_SERVER', '$_COOKIE', '$_SESSION', '$_FILES', '$_REQUEST'];
        
        foreach ($superGlobals as $global) {
            if (str_contains($content, $global)) {
                $this->warnings[] = sprintf(
                    'Class %s uses %s directly, consider using Request object instead',
                    $reflection->getName(),
                    $global
                );
            }
        }
    }
    
    /**
     * 检查单例模式
     */
    private function checkSingletonPattern(\ReflectionClass $reflection): void
    {
        // 检查是否有 getInstance 方法
        if (!$reflection->hasMethod('getInstance')) {
            return;
        }
        
        // 检查是否有静态 $instance 属性
        if (!$reflection->hasProperty('instance')) {
            return;
        }
        
        $property = $reflection->getProperty('instance');
        
        if (!$property->isStatic()) {
            return;
        }
        
        // 检查是否标记为 Resettable
        $attributes = $property->getAttributes(Attribute\Resettable::class);
        
        if (empty($attributes)) {
            $this->warnings[] = sprintf(
                'Singleton class %s::$instance is not marked as #[Resettable], may cause issues in persistent mode',
                $reflection->getName()
            );
        }
    }
    
    /**
     * 扫描目录中的所有类
     */
    public function scanDirectory(string $directory): array
    {
        $results = [];
        
        $iterator = new \RecursiveIteratorIterator(
            new \RecursiveDirectoryIterator($directory)
        );
        
        foreach ($iterator as $file) {
            if ($file->isFile() && $file->getExtension() === 'php') {
                $className = $this->extractClassName($file->getPathname());
                
                if ($className !== null && class_exists($className, true)) {
                    $warnings = $this->checkClass($className);
                    
                    if (!empty($warnings)) {
                        $results[$className] = $warnings;
                    }
                }
            }
        }
        
        return $results;
    }
    
    /**
     * 从文件中提取类名
     */
    private function extractClassName(string $filepath): ?string
    {
        $content = file_get_contents($filepath);
        
        if (!preg_match('/namespace\s+([^;]+);/', $content, $nsMatch)) {
            return null;
        }
        
        if (!preg_match('/class\s+(\w+)/', $content, $classMatch)) {
            return null;
        }
        
        return $nsMatch[1] . '\\' . $classMatch[1];
    }
}
```

## 3. 兼容性适配器

### 3.1 LegacyRequestAdapter

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime\Compat;

use Weline\Framework\Http\Request;
use Weline\Framework\Manager\ObjectManager;

/**
 * 遗留代码请求适配器
 * 
 * 允许遗留代码继续使用超全局变量
 */
final class LegacyRequestAdapter
{
    /**
     * 同步 Request 对象到超全局变量
     */
    public static function syncToGlobals(): void
    {
        $request = ObjectManager::getInstance(Request::class);
        
        $_GET = $request->getGet();
        $_POST = $request->getPost();
        $_SERVER = $request->getServer();
        $_COOKIE = $request->getCookies();
    }
    
    /**
     * 同步超全局变量到 Request 对象
     */
    public static function syncFromGlobals(): void
    {
        $request = ObjectManager::getInstance(Request::class);
        $request->initFromGlobals();
    }
}
```

### 3.2 LegacySessionAdapter

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime\Compat;

use Weline\Framework\Session\SessionInterface;
use Weline\Framework\Manager\ObjectManager;

/**
 * 遗留代码 Session 适配器
 * 
 * 将 SessionInterface 同步到 $_SESSION
 */
final class LegacySessionAdapter
{
    private static bool $synced = false;
    
    /**
     * 启用 $_SESSION 兼容模式
     */
    public static function enable(): void
    {
        if (self::$synced) {
            return;
        }
        
        $session = ObjectManager::getInstance(SessionInterface::class);
        $session->start();
        
        // 同步到 $_SESSION
        $_SESSION = $session->all();
        
        // 注册 shutdown 函数保存
        register_shutdown_function(function() use ($session) {
            foreach ($_SESSION as $key => $value) {
                $session->set($key, $value);
            }
            $session->save();
        });
        
        self::$synced = true;
    }
    
    /**
     * 重置
     */
    public static function reset(): void
    {
        self::$synced = false;
        $_SESSION = [];
    }
}
```

## 4. 运行时兼容性模式

### 4.1 CompatibilityMode

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

/**
 * 兼容性模式
 */
enum CompatibilityMode: string
{
    /**
     * 严格模式：不兼容代码会抛出异常
     */
    case STRICT = 'strict';
    
    /**
     * 警告模式：不兼容代码会记录警告
     */
    case WARNING = 'warning';
    
    /**
     * 兼容模式：自动适配遗留代码
     */
    case COMPAT = 'compat';
    
    /**
     * 静默模式：忽略所有兼容性问题
     */
    case SILENT = 'silent';
}
```

### 4.2 兼容性配置

```php
<?php
// app/etc/env.php
return [
    'runtime' => [
        // 兼容性模式
        'compatibility_mode' => 'compat', // strict, warning, compat, silent
        
        // 自动同步超全局变量
        'sync_globals' => true,
        
        // 自动同步 $_SESSION
        'sync_session' => true,
        
        // 扫描不兼容代码
        'scan_incompatible' => true,
    ],
];
```

## 5. 兼容性中间件

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime\Middleware;

use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Message\ResponseInterface;
use Psr\Http\Server\MiddlewareInterface;
use Psr\Http\Server\RequestHandlerInterface;
use Weline\Framework\Runtime\Compat\LegacyRequestAdapter;
use Weline\Framework\Runtime\Compat\LegacySessionAdapter;
use Weline\Framework\App\Env;

/**
 * 兼容性中间件
 */
final class CompatibilityMiddleware implements MiddlewareInterface
{
    public function process(
        ServerRequestInterface $request, 
        RequestHandlerInterface $handler
    ): ResponseInterface {
        $config = Env::getInstance()->getConfig('runtime') ?? [];
        
        // 同步超全局变量
        if ($config['sync_globals'] ?? true) {
            LegacyRequestAdapter::syncToGlobals();
        }
        
        // 同步 Session
        if ($config['sync_session'] ?? true) {
            LegacySessionAdapter::enable();
        }
        
        return $handler->handle($request);
    }
}
```

## 6. CLI 检测命令

```php
<?php
// bin/w runtime:check

use Weline\Framework\Runtime\CompatibilityChecker;

$checker = new CompatibilityChecker();

// 扫描模块目录
$results = $checker->scanDirectory(APP_CODE_PATH);

if (empty($results)) {
    echo "All classes are compatible with persistent mode.\n";
    exit(0);
}

echo "Found compatibility issues:\n\n";

foreach ($results as $class => $warnings) {
    echo "Class: {$class}\n";
    foreach ($warnings as $warning) {
        echo "  - {$warning}\n";
    }
    echo "\n";
}

exit(1);
```
