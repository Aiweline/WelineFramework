> **预计工作量**: 2-3 天

## 0. 与 WeAsync 的关系

运行时抽象层定义了框架与运行环境的接口，**WeAsync**（照搬 Workerman）作为其中一种运行时实现：

```
RuntimeInterface
├── FpmRuntime        → 传统 PHP-FPM
└── WeAsyncRuntime    → 基于 WeAsync（照搬 Workerman）
```

WeAsync 提供底层的多进程、事件循环、连接管理，而 `WeAsyncRuntime` 负责将其与框架的 `RuntimeInterface` 对接。

---

## 1. 概述

运行时抽象层是实现 FPM/WeAsync 双模式的核心，它将框架核心逻辑与具体运行环境解耦。

### 1.1 设计原则

| 原则 | 说明 |
|------|------|
| **接口隔离** | 只暴露必要的方法 |
| **依赖倒置** | 框架依赖抽象，不依赖具体运行时 |
| **单一职责** | 每个运行时只负责自己的生命周期管理 |

## 2. 核心接口定义

### 2.1 RuntimeInterface 完整定义

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Message\ResponseInterface;
use Psr\Http\Server\RequestHandlerInterface;

/**
 * 运行时接口
 * 
 * 所有运行时适配器必须实现此接口
 */
interface RuntimeInterface
{
    /**
     * 启动运行时
     * 
     * 在 FPM 模式下每个请求都会调用
     * 在常驻模式下只在进程启动时调用一次
     * 
     * 职责：
     * - 加载框架配置
     * - 初始化 DI 容器
     * - 注册服务提供者
     * - 编译路由表
     * - 预热必要的缓存
     */
    public function bootstrap(): void;
    
    /**
     * 处理 HTTP 请求
     * 
     * @param ServerRequestInterface $request PSR-7 请求对象
     * @return ResponseInterface PSR-7 响应对象
     */
    public function handle(ServerRequestInterface $request): ResponseInterface;
    
    /**
     * 请求处理完成后的清理
     * 
     * 在常驻模式下每个请求处理完成后调用
     * 
     * 职责：
     * - 清理请求级实例
     * - 重置可变状态
     * - 释放临时资源
     * - 归还连接到连接池
     */
    public function reset(): void;
    
    /**
     * 终止运行时
     * 
     * 在 FPM 模式下每个请求结束时调用
     * 在常驻模式下只在进程退出时调用
     * 
     * 职责：
     * - 关闭数据库连接
     * - 清理所有资源
     * - 保存必要状态
     */
    public function terminate(): void;
    
    /**
     * 获取运行时类型
     */
    public function getType(): RuntimeType;
    
    /**
     * 是否为常驻内存模式
     */
    public function isPersistent(): bool;
    
    /**
     * 获取请求处理器
     */
    public function getHandler(): RequestHandlerInterface;
    
    /**
     * 获取状态管理器
     */
    public function getStateManager(): StateManagerInterface;
    
    /**
     * 启动服务器（仅常驻模式）
     */
    public function serve(string $host, int $port, int $workers): void;
}
```

### 2.2 AbstractRuntime 基类

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Message\ResponseInterface;
use Psr\Http\Server\RequestHandlerInterface;
use Weline\Framework\App\Env;
use Weline\Framework\Manager\ObjectManager;
use Weline\Framework\Router\Core as Router;

/**
 * 运行时基类
 * 
 * 提供通用实现，子类只需实现特定逻辑
 */
abstract class AbstractRuntime implements RuntimeInterface
{
    protected bool $bootstrapped = false;
    protected StateManagerInterface $stateManager;
    protected RequestHandlerInterface $handler;
    
    public function __construct()
    {
        $this->stateManager = new StateManager();
    }
    
    /**
     * 启动运行时
     */
    public function bootstrap(): void
    {
        if ($this->bootstrapped && $this->isPersistent()) {
            return; // 常驻模式下只启动一次
        }
        
        // 1. 初始化框架
        \Weline\Framework\App::init();
        
        // 2. 加载配置
        $this->loadConfiguration();
        
        // 3. 初始化 DI 容器
        $this->initializeContainer();
        
        // 4. 注册状态管理
        $this->registerStateResettables();
        
        // 5. 预热（仅常驻模式）
        if ($this->isPersistent()) {
            $this->warmup();
        }
        
        $this->bootstrapped = true;
    }
    
    /**
     * 处理请求
     */
    public function handle(ServerRequestInterface $request): ResponseInterface
    {
        // 1. 创建请求上下文
        $context = RequestContext::create();
        
        try {
            // 2. 注入请求到上下文
            $this->injectRequest($request, $context);
            
            // 3. 路由分发
            return $this->dispatch($request);
            
        } finally {
            // 4. 常驻模式下重置状态
            if ($this->isPersistent()) {
                $this->reset();
            }
        }
    }
    
    /**
     * 重置请求级状态
     */
    public function reset(): void
    {
        // 1. 重置状态管理器中的所有可重置对象
        $this->stateManager->resetRequestState();
        
        // 2. 销毁请求上下文
        RequestContext::destroy();
        
        // 3. 清理请求级实例
        ObjectManager::resetRequestInstances();
    }
    
    /**
     * 终止运行时
     */
    public function terminate(): void
    {
        // 1. 重置所有状态
        $this->stateManager->resetAll();
        
        // 2. 关闭连接池
        $this->closeConnectionPools();
        
        // 3. 触发终止事件
        $this->fireTerminateEvent();
    }
    
    public function getStateManager(): StateManagerInterface
    {
        return $this->stateManager;
    }
    
    public function getHandler(): RequestHandlerInterface
    {
        return $this->handler;
    }
    
    /**
     * 加载配置（子类可覆盖）
     */
    protected function loadConfiguration(): void
    {
        Env::getInstance();
    }
    
    /**
     * 初始化 DI 容器（子类可覆盖）
     */
    protected function initializeContainer(): void
    {
        ObjectManager::getInstance();
    }
    
    /**
     * 注册需要重置的状态持有者
     */
    protected function registerStateResettables(): void
    {
        // 注册 ObjectManager
        $this->stateManager->register(ObjectManager::getInstance());
    }
    
    /**
     * 预热缓存（仅常驻模式）
     */
    protected function warmup(): void
    {
        // 预热路由
        ObjectManager::getInstance(Router::class);
    }
    
    /**
     * 注入请求到上下文
     */
    protected function injectRequest(
        ServerRequestInterface $request, 
        RequestContext $context
    ): void {
        $welineRequest = $this->convertRequest($request);
        $context->setInstance(\Weline\Framework\Http\Request::class, $welineRequest);
    }
    
    /**
     * 路由分发
     */
    protected function dispatch(ServerRequestInterface $request): ResponseInterface
    {
        $router = ObjectManager::getInstance(Router::class);
        $result = $router->start();
        
        return $this->convertResponse($result);
    }
    
    /**
     * 关闭连接池
     */
    protected function closeConnectionPools(): void
    {
        // 子类实现
    }
    
    /**
     * 触发终止事件
     */
    protected function fireTerminateEvent(): void
    {
        // 子类实现
    }
    
    /**
     * PSR-7 请求转换为框架 Request
     */
    abstract protected function convertRequest(
        ServerRequestInterface $request
    ): \Weline\Framework\Http\Request;
    
    /**
     * 框架响应转换为 PSR-7 Response
     */
    abstract protected function convertResponse(mixed $result): ResponseInterface;
}
```

## 3. FPM 运行时实现

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime\Adapter;

use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Message\ResponseInterface;
use Nyholm\Psr7\Factory\Psr17Factory;
use Nyholm\Psr7Server\ServerRequestCreator;
use Weline\Framework\Runtime\AbstractRuntime;
use Weline\Framework\Runtime\RuntimeType;

/**
 * FPM 运行时
 * 
 * 传统 PHP-FPM 模式的适配器
 */
final class FpmRuntime extends AbstractRuntime
{
    private Psr17Factory $psr17Factory;
    private ServerRequestCreator $requestCreator;
    
    public function __construct()
    {
        parent::__construct();
        
        $this->psr17Factory = new Psr17Factory();
        $this->requestCreator = new ServerRequestCreator(
            $this->psr17Factory,
            $this->psr17Factory,
            $this->psr17Factory,
            $this->psr17Factory
        );
    }
    
    public function getType(): RuntimeType
    {
        return RuntimeType::FPM;
    }
    
    public function isPersistent(): bool
    {
        return false;
    }
    
    /**
     * FPM 不支持 serve
     */
    public function serve(string $host, int $port, int $workers): void
    {
        throw new \RuntimeException('FPM runtime does not support serve()');
    }
    
    /**
     * FPM 模式入口
     */
    public function run(): void
    {
        $this->bootstrap();
        $request = $this->requestCreator->fromGlobals();
        $response = $this->handle($request);
        $this->emitResponse($response);
        $this->terminate();
    }
    
    protected function convertRequest(
        ServerRequestInterface $request
    ): \Weline\Framework\Http\Request {
        $_SERVER = $request->getServerParams();
        $_GET = $request->getQueryParams();
        $_POST = (array) $request->getParsedBody();
        $_COOKIE = $request->getCookieParams();
        $_FILES = $request->getUploadedFiles();
        
        return \Weline\Framework\Manager\ObjectManager::getInstance(
            \Weline\Framework\Http\Request::class
        );
    }
    
    protected function convertResponse(mixed $result): ResponseInterface
    {
        $response = $this->psr17Factory->createResponse(200);
        
        if (is_string($result)) {
            $body = $this->psr17Factory->createStream($result);
            $response = $response->withBody($body);
        } elseif (is_array($result)) {
            $body = $this->psr17Factory->createStream(json_encode($result));
            $response = $response
                ->withHeader('Content-Type', 'application/json')
                ->withBody($body);
        }
        
        return $response;
    }
    
    private function emitResponse(ResponseInterface $response): void
    {
        http_response_code($response->getStatusCode());
        
        foreach ($response->getHeaders() as $name => $values) {
            foreach ($values as $value) {
                header(sprintf('%s: %s', $name, $value), false);
            }
        }
        
        echo $response->getBody();
    }
}
```

## 4. 运行时工厂

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime;

use Weline\Framework\Runtime\Adapter\FpmRuntime;
use Weline\Framework\Runtime\Adapter\WeAsyncRuntime;

/**
 * 运行时工厂
 */
final class RuntimeFactory
{
    public static function create(?RuntimeType $type = null): RuntimeInterface
    {
        $type ??= self::detectType();
        
        return match($type) {
            RuntimeType::FPM => new FpmRuntime(),
            RuntimeType::WEASYNC => new WeAsyncRuntime(),
            default => new FpmRuntime(),
        };
    }
    
    public static function detectType(): RuntimeType
    {
        // 检测是否在 WeAsync Worker 进程中
        if (getenv('WEASYNC_RUNTIME') === '1') {
            return RuntimeType::WEASYNC;
        }
        
        // 检测是否有 WeAsync Worker 类被加载
        if (class_exists('Weline\\WeAsync\\Worker', false)) {
            return RuntimeType::WEASYNC;
        }
        
        return RuntimeType::FPM;
    }
}
```

## 6. WeAsyncRuntime 实现

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Runtime\Adapter;

use Weline\Framework\Runtime\AbstractRuntime;
use Weline\Framework\Runtime\RuntimeType;
use Weline\WeAsync\Worker;
use Weline\WeAsync\Protocol\Request;

/**
 * WeAsync 运行时
 * 
 * 基于 WeAsync（照搬 Workerman）的常驻内存运行时
 */
final class WeAsyncRuntime extends AbstractRuntime
{
    public function getType(): RuntimeType
    {
        return RuntimeType::WEASYNC;
    }
    
    public function isPersistent(): bool
    {
        return true;
    }
    
    /**
     * 启动 WeAsync 服务器
     */
    public function serve(string $host, int $port, int $workers): void
    {
        $worker = new Worker("http://{$host}:{$port}");
        $worker->count = $workers;
        $worker->name = 'Weline';
        
        // Worker 启动时初始化框架
        $worker->onWorkerStart = function($worker) {
            $this->bootstrap();
            $worker->runtime = $this;
        };
        
        // 处理请求
        $worker->onMessage = function($connection, Request $request) {
            $psrRequest = $request->toPsr7();
            $psrResponse = $this->handle($psrRequest);
            $connection->send($psrResponse);
            $this->reset();
        };
        
        Worker::runAll();
    }
}
```
