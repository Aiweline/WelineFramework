## 当前状态

框架 v2 已使用 `#[Acl]` 属性注解进行权限控制。

## v3 扩展计划

### 路由注解

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Router\Attribute;

use Attribute;

#[Attribute(Attribute::TARGET_CLASS | Attribute::TARGET_METHOD | Attribute::IS_REPEATABLE)]
class Route
{
    public function __construct(
        public string $path,
        public array $methods = ['GET'],
        public ?string $name = null,
        public array $middleware = [],
        public array $defaults = [],
        public array $requirements = [],  // 参数正则约束
        public int $priority = 0          // 路由优先级
    ) {}
}

#[Attribute(Attribute::TARGET_CLASS)]
class RouteGroup
{
    public function __construct(
        public string $prefix = '',
        public array $middleware = [],
        public ?string $domain = null
    ) {}
}

#[Attribute(Attribute::TARGET_METHOD)]
class RouteParam
{
    public function __construct(
        public string $name,
        public string $type = 'string',
        public ?string $pattern = null,
        public mixed $default = null
    ) {}
}
```

### 使用示例

```php
<?php
declare(strict_types=1);

namespace Weline\User\Controller\Backend;

use Weline\Framework\Controller\BackendController;
use Weline\Framework\Router\Attribute\Route;
use Weline\Framework\Router\Attribute\RouteGroup;
use Weline\Framework\Acl\Acl;
use Weline\Framework\Validation\Attribute\Validate;

#[RouteGroup(prefix: '/admin/users', middleware: ['auth', 'admin'])]
#[Acl('Weline_User::management', '用户管理', 'mdi mdi-account-group')]
class User extends BackendController
{
    #[Route('/', methods: ['GET'], name: 'admin.users.index')]
    #[Acl('Weline_User::list', '用户列表', 'mdi mdi-view-list')]
    public function index(): string
    {
        return $this->fetch();
    }
    
    #[Route('/{id}', methods: ['GET'], name: 'admin.users.show', requirements: ['id' => '\d+'])]
    #[Acl('Weline_User::view', '查看用户', 'mdi mdi-eye')]
    public function show(int $id): string
    {
        $user = $this->userRepository->find($id);
        return $this->fetch('show', ['user' => $user]);
    }
    
    #[Route('/', methods: ['POST'], name: 'admin.users.store')]
    #[Acl('Weline_User::create', '创建用户', 'mdi mdi-plus')]
    #[Validate(CreateUserRequest::class)]
    public function store(): array
    {
        $user = $this->userService->create($this->request->getPost());
        return $this->success('User created', ['id' => $user->id]);
    }
    
    #[Route('/{id}', methods: ['PUT', 'PATCH'], name: 'admin.users.update')]
    #[Acl('Weline_User::edit', '编辑用户', 'mdi mdi-pencil')]
    #[Validate(UpdateUserRequest::class)]
    public function update(int $id): array
    {
        $this->userService->update($id, $this->request->getPost());
        return $this->success('User updated');
    }
    
    #[Route('/{id}', methods: ['DELETE'], name: 'admin.users.destroy')]
    #[Acl('Weline_User::delete', '删除用户', 'mdi mdi-delete')]
    public function destroy(int $id): array
    {
        $this->userService->delete($id);
        return $this->success('User deleted');
    }
    
    #[Route('/export', methods: ['GET'], name: 'admin.users.export')]
    #[Acl('Weline_User::export', '导出用户', 'mdi mdi-download')]
    #[Cache(ttl: 300, tags: ['users'])]
    public function export(): Response
    {
        return $this->userService->exportToExcel();
    }
}
```

### 验证注解

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Validation\Attribute;

use Attribute;

#[Attribute(Attribute::TARGET_METHOD | Attribute::TARGET_PROPERTY)]
class Validate
{
    public function __construct(
        public string $requestClass,
        public bool $throwOnFailure = true
    ) {}
}

#[Attribute(Attribute::TARGET_PROPERTY | Attribute::IS_REPEATABLE)]
class Rule
{
    public function __construct(
        public string $rule,
        public ?string $message = null
    ) {}
}

// 使用示例
class CreateUserRequest
{
    #[Rule('required', '用户名不能为空')]
    #[Rule('min:3', '用户名至少3个字符')]
    #[Rule('max:50', '用户名最多50个字符')]
    #[Rule('unique:users,username', '用户名已存在')]
    public string $username;
    
    #[Rule('required')]
    #[Rule('email', '邮箱格式不正确')]
    #[Rule('unique:users,email')]
    public string $email;
    
    #[Rule('required')]
    #[Rule('min:8', '密码至少8个字符')]
    #[Rule('regex:/^(?=.*[A-Z])(?=.*[a-z])(?=.*\d)/', '密码必须包含大小写字母和数字')]
    public string $password;
}
```

### 缓存注解

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Cache\Attribute;

use Attribute;

#[Attribute(Attribute::TARGET_METHOD)]
class Cache
{
    public function __construct(
        public int $ttl = 3600,
        public array $tags = [],
        public ?string $key = null,        // 自定义缓存键
        public array $vary = [],           // 根据参数变化
        public bool $enabled = true
    ) {}
}

#[Attribute(Attribute::TARGET_METHOD)]
class CacheInvalidate
{
    public function __construct(
        public array $tags = [],
        public array $keys = []
    ) {}
}

// 使用示例
class ProductRepository
{
    #[Cache(ttl: 3600, tags: ['products'], vary: ['category'])]
    public function findByCategory(int $category): array
    {
        return $this->query()
            ->where('category_id', $category)
            ->get();
    }
    
    #[CacheInvalidate(tags: ['products'])]
    public function save(Product $product): void
    {
        // 保存后自动清除 products 标签的缓存
    }
}
```

### 事件注解

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Event\Attribute;

use Attribute;

#[Attribute(Attribute::TARGET_CLASS | Attribute::TARGET_METHOD)]
class EventListener
{
    public function __construct(
        public string $event,
        public int $priority = 0,
        public bool $async = false
    ) {}
}

#[Attribute(Attribute::TARGET_CLASS)]
class Subscriber
{
    public function __construct(
        public array $events = []
    ) {}
}

// 使用示例
#[EventListener('user.created', priority: 10)]
class SendWelcomeEmailListener
{
    public function handle(UserCreatedEvent $event): void
    {
        $this->mailer->send(new WelcomeEmail($event->user));
    }
}

#[Subscriber]
class UserSubscriber
{
    #[EventListener('user.created')]
    public function onUserCreated(UserCreatedEvent $event): void {}
    
    #[EventListener('user.updated')]
    public function onUserUpdated(UserUpdatedEvent $event): void {}
    
    #[EventListener('user.deleted', async: true)]
    public function onUserDeleted(UserDeletedEvent $event): void {}
}
```

### 定时任务注解

```php
<?php
declare(strict_types=1);

namespace Weline\Framework\Cron\Attribute;

use Attribute;

#[Attribute(Attribute::TARGET_CLASS | Attribute::TARGET_METHOD)]
class Cron
{
    public function __construct(
        public string $expression,        // cron 表达式
        public ?string $name = null,
        public bool $enabled = true,
        public int $timeout = 3600,       // 超时时间（秒）
        public bool $withoutOverlapping = true
    ) {}
}

// 使用示例
#[Cron('0 * * * *', name: 'hourly_cleanup')]
class HourlyCleanupTask
{
    public function execute(): void
    {
        // 每小时执行
    }
}

class ScheduledTasks
{
    #[Cron('*/5 * * * *')]
    public function syncInventory(): void
    {
        // 每 5 分钟同步库存
    }
    
    #[Cron('0 0 * * *', timeout: 7200)]
    public function dailyReport(): void
    {
        // 每天凌晨生成报表
    }
}
```
